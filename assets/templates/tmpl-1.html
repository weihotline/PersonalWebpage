/* Writing class-like function in Javascript is fun.
 * But do you know how does the magic keyword `this` work?
 */
// Simple class
function Person(name, age) {
    this.name = name;
    this.age = age;
}

var p = new Person("John", 10);
p.name; //=> "John"
p.age;  //=> 10


/* when you use `new` before invoking the function Person,
 * the object `this` is created inside the function. At the end,
 * `this` is returned.
 */
// since `this` is a reserved keyword, I use `that` instead.
function Person(name, age) {
    var that = {};
    that.name = name;
    that.age = age;
    return that;
}

p = Person("Sam", 9);
p.name; //=> "Sam"
p.age;  //=> 9


// What if you forgot to put the word `new` when you instantiate
// an object?
function Cat(name) {
    this.name = name;
}

var c = Cat("Gizmo");
c         //=> undefined because nothing is returned from Cat();
c.name    // TypeError; undefined has no property 'name'
this.name //=> "Gizmo"

/* Why does the last line `this.name` return "Gizmo"?
 * It turns out there is a global object `this`.
 * `this` is the global host object. It refers to the host
 * environment: window for browser; global for nodejs.
 * When Cat("Gizmo") is invoked, the property `name` is added 
 * to the global object `this`. Hence this.name return "Gizmo"
 */

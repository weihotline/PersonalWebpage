// Parent Class
function Parent(name) {
    this.name = name;
}

Parent.prototype.sayMyName = function() {
    console.log(this.name);
}


// Child Class
function Child(name) {
    Parent.call(this, name);
}

// Prototype chaining
// Check if Object.create exists (ECMAScript 5)
if (Object.create === undefined) {
    Object.create = function(parent) {
        function Surrogate() {};
        Surrogate.prototype = parent;
        return new Surrogate();
     }
}

Child.prototype = Object.create(Parent.prototype);

// Initialize
var p = new Parent("John");
var c = new Child("Sam");

p.sayMyName(); //=> "John"
c.sayMyName(); //=> "Sam"


// How about overriding parent's method (Java's 'super')
Child.prototype = Object.create(Parent.prototype, {
    sayMyName: {
        value: function() {
            process.stdout.write("I'm ");
            Parent.prototype.sayMyName.call(this);
         }
    }
});

p = new Parent("John");
c = new Child("Sam");

p.sayMyName(); //=> "John"
c.sayMyName(); //=> "I'm Sam"

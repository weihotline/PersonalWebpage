/* Whenever you assign a variable to another variable,
 * you must be aware of the data type.
 * Javascript is always pass by value.
 */

// When primitive data is passed by value, we get a new copy.
// Primitive data types:
// String
var a = "abc";
var b = a;
b += "def";

a //=> "abc"
b //=> "abcdef"

// Number
var num1 = 1;
var num2 = num1;
num1++;

num1 //=> 2;
num2 //=> 1;



// What about Object? (e.g., Array)
// Let me show you a confusing example.
var arr1 = [ 1, 2, 3 ];
var arr2 = arr1;
arr2.push(4);

arr1 //=> [ 1, 2, 3, 4 ];
arr2 //=> [ 1, 2, 3, 4 ];

/* If it is passed by value, why does modifying arr2 affect arr1?
 * It is because variable arr1 holds the reference to [1, 2, 3].
 * When we assign arr1 to arr2, the value, which contains 
 * the reference, is copied from arr1 to arr2. 
 * Same reference means that they both point to
 * the same object.
 */


// Deep duplicate an array:
// one dimension array:
var arr3 = [ 1, 2, 3 ];
var arr4 = arr3.slice(0);
arr4.push(4);

arr3 //=> [ 1, 2, 3 ]
arr4 //=> [ 1, 2, 3, 4 ]


// multi-demension array:
function deepDup(array) {
    var length = array.length,
        dupArray = [];

    for (var i = 0; i < length; i++) {
        var el = array[i];

        if (Array.isArray(el)) {
            dupArray.push(deepDup(el));
        } else {
            dupArray.push(el);
        }
    }

    return dupArray;
}

var arre1 = [ [ [ 1 ], 2 ], 3 ];
var arre2 = deepDup(arre1);
arre2[0][0].push(4);
arre2[0].push(5);
arre2.push(6);

arre1 //=> [ [ [ 1 ], 2 ], 3 ]
arre2 //=> [ [ [ 1, 4 ], 2, 5 ], 3, 6 ]


// Lastly, let's apply what we learn to a simple problem.
// write a subsets method that returns an array of all subsets.
function subsets(array) {
    if (array.length === 0) {
        return [ [] ]; // empty set
    }

    var oldSubs = subsets(array.slice(1)),
        newSubs = oldSubs.map(function (sub) {
            var newSub = sub.slice(0);
            newSub.push(array[0]);
            return newSub;
        });

    return oldSubs.concat(newSubs);
}

subsets([ 1, 2 ]); //=> [ [], [ 2 ], [ 1 ], [ 2, 1 ] ]

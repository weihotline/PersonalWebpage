/* Writing switch statement could be tricky sometimes.
 * Here are some ways you can write switch statement.
 */

// The normal way:
var animal = "cat";
switch (animal) {
  case "dog":
    console.log("woof!");
    break;
  case "cow":
    console.log("moo!");
    break;
  case "cat":
    console.log("meow!");
    break;
  default:
    console.log("WHAT?");
    break;
} //=> "meow!"



// The fallthrough:
switch (animal) {
  case "cat":
    console.log("meow!");
  case "dog":
    console.log("woof!");
  default:
    console.log("WHAT?");
}
//=> "meow!"
//=> "woof!"
//=> "WHAT?"
/* once it hits the first match,
 * it keeps executing until it hits a break;
 */



// The multi-matches:
switch (animal) {
  case "dog", "cat":
    console.log("Pets!");
    break;
  default:
    console.log("Wild Animals!");
    break;
}//=> "Pets!"



// The conditionals:
var name = "Gizmo";
switch (true) {
  case (animal === "cat" && name === "Gizmo"):
    console.log("My Kitten Gizmo!");
    break;
  case (animal === "dog" && name !== "Gizmo"):
    console.log("My Dog's name is not Gizmo!");
    break;
  default:
    console.log("WHAT?");
}//=> "My Kitten Gizmo!"

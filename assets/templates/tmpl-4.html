// check if a number is an integer.
function isInteger(num) {
    return (num ^ 0) === num;
}
/* num ^ 0 basically flooring the number: Math.floor(num) 
 * Integer should have the same value after being floored.
 */

// Write your isNaN method:
function isTrueNaN(x) {
    return x !== x;
}



// What is going on with the below codes:
for (var i = 0; i < 5; i++) {
    var button = document.createElement('button');
    button.appendChild(document.createTextNode('button ' + i);
    button.addEventListener('click', function() {
        alert(i);
    }

    document.body.appendChild(button);
}
/* Obviously the above codes create 5 buttons on the document
 * body. When you click on each button, you expect to see
 * the corresponding number of each button. But oops, they
 * all alert the number 5 to you. WHY? Because they all share
 * the same i and at the end of the loop, i's value is 5.
 *
 * How to fix this? Give each listener their own copies of i.
 */
for (var i = 0; i < 5; i++) {
    // ...
    (function(i) {
        button.addEventListener('click', function() {
        // The passed i is captured (by closure).
        // It's a different variable than the i used for for-loop
            alert(i);
        }
    })(i);

    // ...
}


// Understand one-line assignment:
(function() {
    var a = b = 3;
})();

console.log(a); //=> undefined
console.log(b); //=> 3
// WHAT? Let's rewrite the above function to another format:
(function() {
    var a = b;
    b = 3;
})();
/* You see the problem? a is a local variable in the IIFE. It is
 * undefined outside the scope. b is a global variable.
 * Hence it is 3;
 */



// What is going on with the below codes:
var list = readAHugeList();
var nextListItem = function() {
    var item = list.pop();

    if (item) {
        // process the item
        nextListItem();
    }
}; // Stack Overflow if the list is way too big.

// I think this solution is a trick:
var nextListItem = function() {
    var item = list.pop();

    if (item) {
        // process the item
        setTimeout(nextListItem, 0);
    }
};
/* How does it work? The idea behind this is called event loop.
 * The recursive call of nextListItem is pushed to the event loop
 * and the function exits (stack frame is clear), and then 
 * the timed recursive call is invoked right after.
 *
 * Recursion is achieved without a direct recursive call.
 */
